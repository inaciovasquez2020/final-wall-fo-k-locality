\section*{Mixed Fixed Part for Admissible Variations}

\begin{definition}[Admissible VMHS]
A variation of mixed Hodge structures with regular singularities at infinity and nilpotent residues.
\end{definition}

\begin{lemma}[Monodromy invariants]
For an admissible VMHS $\mathbb V$, the space of flat sections
\[
H^0(S,\mathbb V_{\mathbb Q})
\]
is stable under the Hodge and weight filtrations.
\end{lemma}

\begin{theorem}[Fixed Part Theorem]
For an admissible VMHS, the monodromy-invariant part defines a rational mixed Hodge substructure.
\end{theorem}

\begin{corollary}[Rationality of horizontal subtori]
Horizontal real subtori coincide with their rational envelopes.
\end{corollary}
