\begin{abstract}
We define a Terminal Rigidity Witness as a physical system with nontrivial global structure but zero operational information capacity.
We show that the classical Einstein--Rosen bridge in the maximally extended Schwarzschild spacetime is a canonical example under causal accessibility.
\end{abstract}

\section{Terminal rigidity}
\begin{definition}
A physical system $S$ is a Terminal Rigidity Witness if $\Capacity(\mathcal C_S)=0$ while $S$ has nontrivial global structure.
\end{definition}
\section{Einstein--Rosen bridge}
\begin{axiom}[Causal separation across the bridge]
No causal curve connects the two asymptotic regions of the maximally extended Schwarzschild spacetime.
\end{axiom}
\begin{theorem}
The Einstein--Rosen bridge is a Terminal Rigidity Witness.
\end{theorem}
\begin{proof}
By causal separation, no operational interface restricted to causal propagation can transmit influence from hidden states on one asymptotic region to transcripts observed on the other.
Hence for every $T$ and every admissible strategy, $I(X;Y_{1:T})=0$, so $\TC(T)=0$ and thus $\Capacity(\mathcal C_{\mathrm{ERB}})=0$.
The maximally extended spacetime nevertheless has nontrivial global structure.
\end{proof}
